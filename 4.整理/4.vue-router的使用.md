# vue-router的使用

## 1.路由的懒加载

```js
// router.index.js
import Vue from 'vue'
import Router from 'vue-router'

Vue.use(Router)

export default new Router({
  routes: [
    {path: '/',redirect: '/user'},
    {path: '/register',name: 'register',component: ()=>import('../pages/Register')},
    {path: '/login',name: 'login',component: ()=>import('../pages/Login')},
    {path: '*',name: 'notfound',component: ()=>import('../pages/NotFound')},
    {path: '/user',component: ()=>import('../pages/layout'),
      meta: {
        requiresAuth: true
      },
      children: [

      ]
    }
  ]
})
```

## 2.路由拦截

```js
// utils/permission.js
import router from '../router'

router.beforeEach((to, from, next) => {
  if (to.matched.some(record => record.meta.requiresAuth)) {
    if (!sessionStorage.getItem("key")) {
      next({
        path: '/login',
        query: {
          redirect: to.fullPath
        }
      })
    } else {
      next()
    }
  }else {
    next()
  }
})
```

## 4.路由传参

- 使用params时，当页面刷新数据会丢失
- 推荐使用query

```js

// template
<router-link :to="{path:'detal',query:'{id:1}'}"> 前往detail页面 </router-link>

// script
this.$router.push({
  path:'detail',
  query: {
    id:'1'
  }
})
// 获取参数
this.$route.query.id
```
